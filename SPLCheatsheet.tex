\headerbox{Lines in Dialogs}{name=lines1,column=1,row=0,bottomaligned=layout}{


\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Example 1}}

\begin{itemize}\compresslist
\item Hamlet:\\
 You lying stupid fatherless big smelly half-witted coward!
\item 2*2*2*2*2*2*(-1) = -32
 \end{itemize}

\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Constants }}
\begin{itemize}\compresslist
 \item Positive and neutral Nouns equal 1
 \item Negative nouns equal -1
 \item Adjectives multiply their noun with 2
 \item Feelings of nouns and adjectives must match
 \end{itemize}



\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Example 2}}
\begin{itemize}\compresslist
\item You are as stupid as the difference between a handsome rich brave hero and 
thyself!
\item Julia = 8-Julia
\end{itemize}

\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Assignment }}
\begin{itemize}\compresslist
 \item ...are as<any adjective> as <operator> <constant/reference/character> and <constant/reference/character>
\end{itemize}

\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Operators and References }}
\begin{itemize}\compresslist
\item Operators: sum of,difference between,product of, quotient of,square of, cube of 
 \item References: yourself, thyself, myself
 \end{itemize}
 
 

}

\headerbox{Lines in Dialogs}{name=lines2,column=2,row=0,bottomaligned=layout}{

 \colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Example 3 }}
\begin{itemize}\compresslist
\item  Juliet:\\
 Am I better than you?\\
Hamlet:\\
 If so, let us proceed to scene III.\\
 \end{itemize}
 
 \colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Logical Expressions }}
\begin{itemize}\compresslist
 \item is <X> as good as <Y> tests for X == Y
 \item is <X> better than <Y> tests for X > Y
 \item is <X> worse than <Y> tests for X < Y
 \end{itemize}

\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Conditional GoTo }}
\begin{itemize}\compresslist
 \item Logical Expression
 \item <If so/If not>, <Let us, we shall, we must> <return to/ proceed to> <SceneX>
 \end{itemize}

\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - In- and Output }}
\begin{itemize}\compresslist
 \item In: Listen to your heart.
 \item Out: Speak your mind.
 \end{itemize}

\colorbox[HTML]{CCFFFF}{\makebox[\textwidth-2\fboxsep][l]{\bf - Stack Operations }}
\begin{itemize}\compresslist
 \item Push: \\Lady Macbeth:\\
 Remember <expression>.
 \item Pop: \\Lady Macbeth:\\
 Recall <Comment>
 \end{itemize}


}
